MVC Webapp Architecture

  • client request a view
  • client invokes an action (change view, submit form)
  • server process the request
  • server sends a new view to the client

Server role

  • authentication&authorization

  • business logic
  • HTML generation (including dates, timezones, numbers, etc)
  • batch jobs

  • render HTML page

Deployment model

  • one .war per webapp

  • one JVM per servlet container

Perfectly reasonable architectures but...

What happens when...

  • a CSS attribute must be changed

  • internal server error

// hiding how we store the auth data
app.service('AuthContext', function($localStorage) {
  this.put = function(authentication) {
    localStorage.authentication = authentication;
  };
  this.reset = function() {
    delete $localStorage.authentication;
  };
  this.get = function() {
    return $localStorage.authentication;
  };
  var authentication = this.get();
  if (authentication) {
     $log.debug("auth has been restored from local storage");
  }
  return this;
});

// put 'Authorization' header on each outgoing request
$httpProvider.interceptors.push(function($rootScope, $q, $log, AuthContext){
  $log.info("enabling auth interceptor");
  return {
    'request' : function(request) {
       var auth = AuthContext.get();
       if (auth) {
         request.headers.Authorization = auth.token;
       }
       return request;
    }
  }
});

Session timeout replacement

// postpone session expiration until a 'session.keepalive' event is produced
app.run(function($rootScope, $state, $log, $timeout, AuthContext) {
  var timeout = moment.duration(10, 'minutes');
  $log.info("session timeout is " + timeout.asMinutes() + " minute(s)");
  function relogin() {
    $log.warn("closing session by inactivity");
    AuthContext.reset();
    $state.go('relogin', {
      reason : 'Your session has timed out, please login again'
    });
  }
  var promise = $timeout(relogin, timeout.asMilliseconds());
  $rootScope.$on("session.keepalive", function() {
    $timeout.cancel(promise);
    promise = $timeout(relogin, timeout.asMilliseconds());
  });
});
					
// a 'session.keepalive' event producer
$httpProvider.interceptors.push(function($rootScope) {
  return {
    'request' : function(request) {
      $rootScope.$broadcast("session.keepalive");
      return request;
    }
  };
});

Logout from all tabs

// global login/logout via 'storage' event
app.run(function($window, $state, $log) {
  $log.info("installing global login/logout listener");
  var authKey = 'ngStorage-authentication';
  $window.addEventListener('storage', function(evt) {
    if (evt.key === authKey && evt.newValue === null) {
      $log.info("seen a logout in another tab, redirecting to login");
      $state.go('login');
    }
    if (evt.key === authKey && evt.newValue !== null) {
      $log.info("seen a login in another tab, redirecting to home");
      $state.go('home');
    }
  });
});
